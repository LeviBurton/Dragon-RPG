tree "Root"
	fallback
		tree "Chase"
		tree "Attack"
		tree "FindCombatTarget"
		tree "Idle"

tree "Attack"
	while 
		sequence
			HasTarget
			TargetIsAlive
			IsWithinWeaponRange 
		
		sequence 
			StopMoving
			WaitForRecovery
			Attack

tree "FindCombatTarget"
	FindCombatTarget

tree "Chase"
	while 
		sequence
			HasTarget
			TargetIsAlive
			not IsWithinWeaponRange

		sequence
			MoveToTargetPosition
			StopMoving

tree "Idle"
	while 
		sequence
			not FindCombatTarget

		sequence
			SetTargetToHome
			MoveToTargetPosition
			SetSpeed 1.0
			StopMoving
			Idle
