tree "Root"
	fallback
		tree "ReturnHome"
		tree "Chase"
		tree "Attack"
		tree "FindCombatTarget"
		tree "Idle"

tree "ReturnHome"
	sequence
		AllHeroesDead
		SetTargetToHome
		MoveToTargetCursor		
		StopMoving

tree "Attack"
	while 
		sequence
			HasTarget
			TargetIsAlive
			IsWithinWeaponRange
			StopMoving
			LookAtTarget

		sequence 
		
			WaitForRecovery
			Attack

tree "FindCombatTarget"
	FindCombatTarget

tree "Chase"
	while 
		sequence
			HasTarget
			TargetIsAlive
			not IsWithinWeaponRange

		sequence
			MoveToTarget
			StopMoving

tree "Idle"
	while 
		sequence
			not FindCombatTarget

		sequence
			Idle
