tree "Root"
	fallback
		not IsAlive
		tree "ReturnHome"
		tree "Chase"
		tree "Attack"
		tree "FindClosestHero"
		tree "Idle"

tree "ReturnHome"
	sequence
		AllHeroesDead
		SetTargetToHome
		SetWalking
		MoveToTargetCursor
		StopMoving

tree "Attack"
	while 
		sequence
			IsAlive
			HasTarget
			TargetIsAlive
			IsWithinWeaponRange
			StopMoving
			LookAtTarget

		sequence 
			WaitForRecovery
			Attack

tree "FindClosestHero"
	FindClosestHero

tree "Chase"
	while 
		sequence
			IsAlive
			HasTarget
			TargetIsAlive
			not IsWithinWeaponRange

		sequence
			MoveToTarget
			StopMoving

tree "Idle"
	while 
		sequence
			IsAlive
			not FindClosestHero

		sequence
			Idle
