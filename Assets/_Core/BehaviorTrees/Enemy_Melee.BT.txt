tree "Root"
	fallback
		tree "Attack"
		not tree "TargetAquire"
		tree "Chase"
		tree "Idle"

tree "Attack"
	while 
		sequence
			HasTarget
			TargetIsAlive
			InAttackRange 
		
		sequence 
			StopMoving
			WaitForRecovery
			Attack

tree "TargetAquire"
	Target_Acquire

tree "Chase"
	while 
		sequence
			MoveToMelee
			TargetIsAlive
			not InAttackRange
		
		sequence
			MoveToTargetPosition
			StopMoving
tree "Idle"
	sequence
		fallback
			Target_Acquire
