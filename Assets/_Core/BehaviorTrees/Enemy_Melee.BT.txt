tree "Root"
	fallback
		tree "Chase"
		tree "Attack"
		tree "FindCombatTarget"
		tree "Idle"

tree "Attack"
	while 
		sequence
			HasTarget
			TargetIsAlive
			InAttackRange 
		
		sequence 
			StopMoving
			WaitForRecovery
			Attack

tree "FindCombatTarget"
	FindCombatTarget

tree "Chase"
	while 
		sequence
			HasTarget
			TargetIsAlive
			not InAttackRange

		sequence
			MoveToTargetPosition
			StopMoving

tree "Idle"
	while 
		sequence
			not FindCombatTarget

		sequence
			Idle
